cmake_minimum_required(VERSION 2.8.3 )
project(ttb_expressions)
 
## Use c++ 11x std 
set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS}")
 
find_package(catkin REQUIRED COMPONENTS 
  roscpp 
  system_config 
  fsystem 
  autodiff 
  event_handling 
  alica_engine 
  ttb_worldmodel
  ttb_robot 
  system_util 
  ecl_geometry 
  kobuki_dock_drive 
  kdl_conversions 
  robot_control
  ttb_msgs
  sound_play
  asp_commons
  asp_solver_wrapper
  asp_solver
  alica_dummy_solver
) 

catkin_package(
  INCLUDE_DIRS autogenerated/include
  LIBRARIES ttb_expressions
  CATKIN_DEPENDS alica_engine system_config ttb_worldmodel kobuki_dock_drive ecl_geometry kdl_conversions asp_commons asp_solver_wrapper asp_solver ttb_robot alica_dummy_solver
)
  
include_directories(autogenerated/include ${catkin_INCLUDE_DIRS})

file(GLOB_RECURSE autogen_SOURCES "autogenerated/src/*.cpp")
add_library(ttb_expressions ${autogen_SOURCES};) 

target_link_libraries(ttb_expressions ${catkin_LIBRARIES})
